// Creating a new XMLHttpRequest object for get Request using js var xhr = new XMLHttpRequest(); // Configuring the request: GET method and URL xhr.open('GET', 'https://example.com/data', true); // Setting up a callback function to handle the response xhr.onreadystatechange = function() { if (xhr.readyState == 4 && xhr.status == 200) { // Processing the response console.log(xhr.responseText); } }; // Sending the request xhr.send(); ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------ // Creating a new XMLHttpRequest object for post Request using js var xhr = new XMLHttpRequest(); // Configuring the request: POST method and URL xhr.open('POST', 'https://example.com/data', true); // Setting the Content-Type header for a POST request xhr.setRequestHeader('Content-Type', 'application/json'); // Setting up a callback function to handle the response xhr.onreadystatechange = function() { if (xhr.readyState == 4 && xhr.status == 200) { // Processing the response console.log(xhr.responseText); } }; // Data to be sent in the request body var data = { key: 'value' }; // Sending the request with data xhr.send(JSON.stringify(data)); ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- // Making a GET request using jQuery $.ajax({ url: 'https://example.com/data', method: 'GET', success: function(response) { // Handling successful response console.log(response);
